const yaml = require('js-yaml');
const fs = require('fs');

const VALID_KINDS = ['单行文本','多行文本','单选','多选','数字','日期','成员','级联单选','级联多选','进度','评分','链接','引用'];
const VALID_GROUPS = ['需求','任务','事务'];
const OPTION_KINDS = ['单选','多选','级联单选','级联多选'];

// 取出含 "types:" 的第一个 ```yaml 围栏块内容；无则 null
function extractTypesBlock(mdText) {
  const re = /```ya?ml\s*\n([\s\S]*?)\n```/g;
  let m;
  while ((m = re.exec(mdText)) !== null) {
    if (/(^|\n)\s*types\s*:/.test(m[1])) return m[1];
  }
  return null;
}

function parseVertical(mdText) {
  const block = extractTypesBlock(mdText);
  if (!block) return { types: [] };
  const obj = yaml.load(block) || {};
  if (!Array.isArray(obj.types)) obj.types = [];
  return obj;
}

module.exports = { extractTypesBlock, parseVertical, VALID_KINDS, VALID_GROUPS, OPTION_KINDS };
````

// 返回错误字符串数组；空数组=通过
function validateVertical(parsed) {
  const errors = [];
  const types = (parsed && Array.isArray(parsed.types)) ? parsed.types : [];
  types.forEach((t, i) => {
    const label = t && t.name ? t.name : `类型#${i + 1}`;
    if (!t || !t.name) errors.push(`${label}: 缺 name`);
    if (t && t.group && !VALID_GROUPS.includes(t.group)) {
      errors.push(`${label}: group "${t.group}" 非法（需 ${VALID_GROUPS.join('/')}）`);
    }
    const fields = (t && Array.isArray(t.fields)) ? t.fields : [];
    fields.forEach((f, j) => {
      const flabel = `${label}.${f && f.name ? f.name : `字段#${j + 1}`}`;
      if (!f || !f.name) errors.push(`${flabel}: 字段缺 name`);
      if (!f || !VALID_KINDS.includes(f.kind)) errors.push(`${flabel}: kind "${f && f.kind}" 非法（需 13 种属性类型之一）`);
      if (f && OPTION_KINDS.includes(f.kind) && (!Array.isArray(f.options) || f.options.length === 0)) {
        errors.push(`${flabel}: ${f.kind} 缺 options`);
      }
    });
  });
  return errors;
}
````

// CLI: node vertical_template.js <file.md> [更多文件...]；无参则校验 references/verticals/*.md
if (require.main === module) {
  const path = require('path');
  let files = process.argv.slice(2);
  if (files.length === 0) {
    const dir = path.join(__dirname, '..', 'references', 'verticals');
    files = fs.existsSync(dir)
      ? fs.readdirSync(dir).filter(f => f.endsWith('.md') && !f.startsWith('_')).map(f => path.join(dir, f))
      : [];
  }
  let bad = 0;
  for (const f of files) {
    const errs = validateFile(f);
    if (errs.length) { bad++; console.log(`❌ ${path.basename(f)}`); errs.forEach(e => console.log('   - ' + e)); }
    else console.log(`✅ ${path.basename(f)}`);
  }
  process.exit(bad ? 1 : 0);
}

````markdown
# 行业结构模板（verticals）

每个 `<行业>.md` = 一个行业的完整结构包。搭建时按客户「行业」匹配（命中→加载 / 未命中长尾→现场调研生成 / 都不沾边→降级通用 WBS）。

## 文件格式
- **散文区**（人编辑 / LLM 读）：项目类型、阶段门流程、测试维度、标准、术语、客户关注点。
- **结构化区**：一个 ` ```yaml ` 块，`types:` 列出专属工作项类型与字段，供执行层精确建出。
- 文件**首行注释**写 synopsis（别名关键词），供 LLM 就近匹配：
  `<!-- synopsis: 汽车电子 / 车载 / ADAS / 智能座舱 / 主机厂 / Tier1 -->`

## YAML 约束（必须，否则校验不过）
- `group` ∈ `需求 / 任务 / 事务`
- 字段 `kind` ∈ 13 种 PingCode 属性类型：单行文本 / 多行文本 / 单选 / 多选 / 数字 / 日期 / 成员 / 级联单选 / 级联多选 / 进度 / 评分 / 链接 / 引用
- `单选 / 多选 / 级联单选 / 级联多选` 必须带 `options: [...]`
- 每个 type 必须有 `name`，每个 field 必须有 `name`

## 校验
`node scripts/vertical_template.js`           # 校验全部模板
`node scripts/vertical_template.js 汽车电子.md` # 校验单个

## 新增 / 沉淀一个行业
1. 复制现有 `.md` 改内容；或由搭建时长尾生成 → 交付时确认沉淀。
2. 跑校验通过。
3. 提交 GitHub，同事 `git pull` 即得。

## 专属类型怎么选（重要）
优先**补 PingCode 短板**：用行业标准对照 PingCode 能力，缺口处正是最该建的自定义类型
（如 ASPICE `SUP.10 变更请求` PingCode 不满足 → 建「变更请求」），既补功能又是演示讲合规的抓手。
````
